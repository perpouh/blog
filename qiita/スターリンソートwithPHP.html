まだPHPで書いてる人が見当たらなかったので。居たらごめんなさい。

<?php
$array = [1,3,4,2,3,7,3,9];
print_r(stalin_sort($array));

function stalin_sort($arr){
	$ret[] = $max = $arr[0];
	foreach($arr as $el){
		if($el <= $max) continue;
		$ret[] = $max = $el;
	}
	return $ret;
}
